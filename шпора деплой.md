## base conf

{ <br>

    server {
      listen 80; # Принимаем запросы с порта 80<br>
      server_name domainname.students.nomoreparties.co; # Указываем название домена<br>

      location / {<br>
                proxy_pass http://localhost:3000; # Перенаправляем запросы на порт 3000<br>
                proxy_http_version 1.1; # Задаёт версию HTTP для прокси<br>
                proxy_set_header Upgrade $http_upgrade; # Необходим для проксирования WebSocket-соединений<br>
                proxy_set_header Connection 'upgrade';  # Необходим для проксирования WebSocket-соединений <br>
                proxy_set_header Host $host; # Содержит адрес домена, на который пришёл запрос<br>
                proxy_cache_bypass $http_upgrade;<br>
      }
    }
    /*

    Description=Docker Compose Application
    Requires=docker.service
    After=docker.service

    [Service]
    Type=oneshot
    RemainAfterExit=yes
    User=truenoob
    Group=docker
    WorkingDirectory=/home/truenoob/app
    ExecStart=/usr/bin/docker-compose up --build -d
    ExecStop=/usr/bin/docker-compose down
    StandardOutput=journal

    [Install]
