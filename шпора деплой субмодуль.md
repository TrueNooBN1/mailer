# обновление git

cd src/email-module<br>
git pull origin main<br>
cd ../..<br>
git add src/email-module<br>
git commit -m "Update email-module to latest"<br>

или

git submodule update --remote src/email-module

# Клонирование 
git clone --recursive <url>

или

git submodule update --init --recursive

## термоядерный вариант

### 1. Клонируем репозиторий вручную
git clone https://github.com/repository backend/src/email-module

### 2. Добавляем его как сабмодуль вручную
git submodule add --force https://github.com/repository backend/src/email-module

# CI/CD
- name: Checkout
  uses: actions/checkout@v3
  with:
    submodules: recursive
